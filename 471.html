import pandas as pd

id = pd.read_table(“mdpout_descriptors.txt”,sep=” +”,engine=’python’)
name2=id[‘snapshot’]
time=id[‘pock_volume’]

#method1
pd.concat([name2,time],axis=1).to_csv(‘volume.txt’,sep=” “,index=False)
